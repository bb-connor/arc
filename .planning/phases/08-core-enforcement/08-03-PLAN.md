a. `TokenBucket` struct (private):
```rust
use std::time::Instant;

struct TokenBucket {
    capacity: f64,
    tokens: f64,
    refill_rate: f64,  // tokens per second
    last_refill: Instant,
}

impl TokenBucket {
    fn new(capacity: f64, refill_rate: f64) -> Self {
        Self {
            capacity,
            tokens: capacity,
            refill_rate,
            last_refill: Instant::now(),
        }
    }

    fn try_consume(&mut self, amount: f64) -> bool {
        self.refill();
        if self.tokens >= amount {
            self.tokens -= amount;
            true
        } else {
            false
        }
    }

    fn refill(&mut self) {
        let elapsed = self.last_refill.elapsed().as_secs_f64();
        self.tokens = (self.tokens + elapsed * self.refill_rate).min(self.capacity);
        self.last_refill = Instant::now();
    }
}
```

d. `Guard` implementation:
```rust
impl Guard for VelocityGuard {
    fn name(&self) -> &str {
        "velocity"
    }

    fn evaluate(&self, ctx: &GuardContext) -> Result<Verdict, KernelError> {
        // Use matched_grant_index from context if available, else default to 0
        let grant_index = ctx.matched_grant_index.unwrap_or(0);
        let key = (ctx.request.capability.id.clone(), grant_index);

        // Check invocation rate limit
        if let Some(max_inv) = self.config.max_invocations_per_window {
            let capacity = max_inv as f64 * self.config.burst_factor;
            let refill_rate = max_inv as f64 / self.config.window_secs.max(1) as f64;

            let mut buckets = self.invocation_buckets.lock()
                .map_err(|_| KernelError::Internal("velocity guard invocation lock poisoned".to_string()))?;
            let bucket = buckets.entry(key.clone())
                .or_insert_with(|| TokenBucket::new(capacity, refill_rate));
            if !bucket.try_consume(1.0) {
                return Ok(Verdict::Deny);
            }
        }

        // Check spend rate limit (if applicable)
        if let Some(max_spend) = self.config.max_spend_per_window {
            let capacity = max_spend as f64 * self.config.burst_factor;
            let refill_rate = max_spend as f64 / self.config.window_secs.max(1) as f64;

            let mut buckets = self.spend_buckets.lock()
                .map_err(|_| KernelError::Internal("velocity guard spend lock poisoned".to_string()))?;
            let bucket = buckets.entry(key)
                .or_insert_with(|| TokenBucket::new(capacity, refill_rate));
            // Spend velocity consumes units based on request context
            // For now, consume 1.0 -- the kernel will pass actual cost in Phase 8 integration
            if !bucket.try_consume(1.0) {
                return Ok(Verdict::Deny);
            }
        }

        Ok(Verdict::Allow)
    }
}
```
